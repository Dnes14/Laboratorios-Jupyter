In[11]:
# cargar tareas y duraciones
for i in tareas:
    p.add(Node(i[0],duration=i[1]["duracion"]))

# cargar al proyecto sus dependencias
for j in dependencias:
    p.link(j[0],j[1])

#actualizar el proyecto
p.update_all()
